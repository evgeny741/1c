//Убрать из строки все цифры
//Нужно разработать программный код, который удалит из строки, которую
//вводит пользователь, все цифры и двойные пробелы.
//Пример:
//Исходная строка: " Это стро4ка с 8 цифрами 7"
//Результат: "Это строка с цифрами"      

Функция УбратьВсеЦифры(СтрА) 
	СтрокаПроверки = СокрЛП(СтрА); 
	ДлинаСтроки = СтрДлина(СтрокаПроверки);  
	
	НовСтрока = "";
	ПравильныеСимволы = " QWERTYUIOPASDFGHJKLZXCVBNMqwertyuiopasdfghjklzxcvbnmЙЦУКЕНГШЩЗХЪФЫВАПРОЛДЖЭЯЧСМИТЬБЮйцукенгшщзхъфывапролджэячсмитьбю";
	Для Сч = 1 по СтрДлина(СтрокаПроверки) Цикл
		ТекСимв = Сред(СтрокаПроверки, Сч, 1);
		Если Найти(ПравильныеСимволы, ТекСимв) > 0 Тогда
			НовСтрока = НовСтрока + ТекСимв;
		КонецЕсли;
	КонецЦикла;
	
	СтрокаПроверки = НовСтрока;    	
		
	Пробел=" ";
	Пока Найти(СтрокаПроверки,Пробел+Пробел) <> 0 Цикл
		СтрокаПроверки = СтрЗаменить(СтрокаПроверки,Пробел+Пробел,Пробел);
	КонецЦикла;
	
	Возврат СтрокаПроверки;
КонецФункции    

Сообщить(УбратьВсеЦифры("   77    Это  стро4ка с 38 цифрами 77"));
