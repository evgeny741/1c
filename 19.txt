	РезультатЦикла = "";

		Символ = Сред(СтрокаПроверки, Сч, 1);

			 РезультатЦикла = РезультатЦикла + Символ;
		КонецЕсли;
	КонецЦикла; 	
	
	Возврат(РезультатЦикла);
КонецФункции
