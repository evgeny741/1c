//Пользователь вводит сумму в рублях в виде числа. Нужно показать
//пользователю представление суммы прописью .
//При этом, если дробная часть числа меньше чем 0.11, то её нужно
//отображать прописью, если дробная часть больше чем 0.11, то её нужно
//отображать числом.
//Примеры:
//Число: 120.07
//Результат: Сто двадцать рублей семь копеек

Функция ПредставлениеСуммыПрописью(ЧислоА) 
	Если 100*(ЧислоА - Окр(ЧислоА,0,0))<11 Тогда       
		Возврат ЧислоПрописью(ЧислоА,"Л=ru_RU; ДП=Ложь","рубль, рубля, рублей, м, копейка, копейки, копеек, ж" );
	Иначе  
		Возврат "" + ЧислоПрописью(ЧислоА,"Л=ru_RU; ДП=Ложь","рубль, рубля, рублей, м" )+ "коп.";	
	КонецЕсли;
КонецФункции;
