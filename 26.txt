//Преобразовать сумму строкой в число
//Нужно из строки вида "x руб. y коп." получить сумму в виде числа и
//сообщить пользователю.
//Пример:
//Исходная строка: 127 руб. 25 коп.
//Результат (число): 127.25

ПоискКопеек = " коп.";
РезультирующаяСтрока = СтрЗаменить(ИсходнаяСтрока,ПоискРублей,"."); 
РезультирующаяСтрока = СтрЗаменить(РезультирующаяСтрока,ПоискКопеек,"");
