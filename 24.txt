//Количество рабочих и выходных дней
//Нужно написать программный код, который вычислит количество рабочих и
//выходных дней между двумя датами и сообщит пользователю.
//Рабочим считается день с понедельника по пятницу. Выходной - суббота или
//воскресенье.
//Пример:
//НачалоПериода = 23.03.2020;
//КонецПериода = 28.03.2020;
//Результат:
//Количество рабочих дней: 5
//Количество выходных дней: 1        

Функция ПеревернутьДату(СтрА)
	Возврат Дата (Прав (СтрА, 4), Сред (СтрА, 4, 2), Лев (СтрА, 2))	
КонецФункции       

НачалоПериода = ПеревернутьДату("23.03.2020");  
КонецПериода = ПеревернутьДату("28.03.2020");  

КолРабДней=0;
КолВыхДней=0;   
       
ПеремДень=НачалоПериода;
Пока ПеремДень <= КонецПериода Цикл
    Если ДеньНедели(ПеремДень )<6 Тогда 
    	КолРабДней=КолРабДней+1;  
	Иначе
		 КолВыхДней = КолВыхДней + 1;
    КонецЕсли; 
    ПеремДень = ПеремДень + 24*3600; 
КонецЦикла;  

Сообщить("Результат:"); 
Сообщить("Количество рабочих дней: " + КолРабДней);
Сообщить("Количество выходных дней: " + КолВыхДней);
