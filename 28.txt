//Среднее арифметическое = Сумма чисел / Количество чисел

Функция ГенераторСлучайныхЧисел(НачЧисло,КонЧисло)
	ГСЧ = Новый ГенераторСлучайныхЧисел();
	СлучайноеЧисло = ГСЧ.СлучайноеЧисло(НачЧисло, КонЧисло);  
	Возврат СлучайноеЧисло;
КонецФункции  

Функция ЗаполнениеМассива(ЧислоЭлементов,НачЧисло,КонЧисло)  
	МассивЧисел = Новый Массив(); 
	Сч = 0;
	Пока Сч < ЧислоЭлементов Цикл  
		СлучайноеЧисло = ГенераторСлучайныхЧисел(НачЧисло,КонЧисло); 
		МассивЧисел.Добавить(СлучайноеЧисло);
		Сч = Сч + 1;     
	КонецЦикла;                       
	Возврат МассивЧисел;
КонецФункции 

Функция ВычислениеСреднегоЗначенияМассива(МассивЧисел,ЧислоЭлементов)  
	СуммаЧисел = 0;
	Для Сч = 0 по МассивЧисел.ВГраница() Цикл   
		СуммаЧисел = СуммаЧисел + МассивЧисел.Получить(Сч);      	
	КонецЦикла;    
	СреднееЗначениеМассива = СуммаЧисел / ЧислоЭлементов;
	Возврат СреднееЗначениеМассива;
КонецФункции

ЧислоЭлементовМассива = 12;
НачЧислоГенератора = 0;
КонЧислоГенератора = 50;  
МассивЧисел = ЗаполнениеМассива(ЧислоЭлементовМассива,НачЧислоГенератора,КонЧислоГенератора);
СреднееЗначениеЧиселМассива = ВычислениеСреднегоЗначенияМассива(МассивЧисел,ЧислоЭлементовМассива);
Сообщить("Среднее значение чисел массива из " + ЧислоЭлементовМассива + " чисел равно " + Цел(СреднееЗначениеЧиселМассива));
